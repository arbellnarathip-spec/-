async function loadData(){
  document.getElementById('tableBody').innerHTML='<tr class="loading-msg"><td colspan="9">⏳ กำลังโหลดข้อมูล...</td></tr>';
  allData=[];
  if(activeDept===0){
    const r=await Promise.all(CONFIG.DEPTS.slice(1).map((d,i)=>fetchSheet(d.sheetId,d.sheetName,i+1)));
    allData=r.flat().sort((a,b)=>(a.rawDate||0)-(b.rawDate||0));
  } else {
    const d=CONFIG.DEPTS[activeDept];
    allData=await fetchSheet(d.sheetId,d.sheetName,activeDept);
  }
  updateHeader(); updateSummary(); renderTable();
}

async function fetchSheet(sheetId,sheetName,deptIdx){
  if(!sheetId||sheetId.startsWith('YOUR')) return [];
  try{
    const res=await fetch(`https://docs.google.com/spreadsheets/d/${sheetId}/gviz/tq?tqx=out:json&sheet=${encodeURIComponent(sheetName)}`);
    const text=await res.text();
    const json=JSON.parse(text.substring(47,text.length-2));
    return (json.table.rows||[]).map((row,i)=>{
      const c=row.c;
      const rd=parseGDate(c[0]?.v);
      return {idx:i+1,rawDate:rd?rd.getTime():0,date:rd?rd.toLocaleDateString('th-TH',{year:'numeric',month:'short',day:'numeric'}):'-',
        type:c[1]?.v||'',name:c[2]?.v||'',income:parseFloat(c[3]?.v)||0,expense:parseFloat(c[4]?.v)||0,
        balance:parseFloat(c[5]?.v)||0,slip:c[6]?.v||'',note:c[7]?.v||'',deptIdx};
    }).filter(r=>r.name);
  }catch{return [];}
}

function parseGDate(v){
  if(!v)return null;
  if(typeof v==='string'&&v.startsWith('Date(')){const p=v.slice(5,-1).split(',');return new Date(+p[0],+p[1],+p[2]);}
  return null;
}

function updateSummary(){
  const ti=allData.reduce((s,r)=>s+r.income,0);
  const te=allData.reduce((s,r)=>s+r.expense,0);
  const lb=allData.length?allData[allData.length-1].balance:0;
  document.getElementById('sumIncome').textContent='฿'+fmt(ti);
  document.getElementById('sumExpense').textContent='฿'+fmt(te);
  document.getElementById('sumBalance').textContent='฿'+fmt(lb);
  document.getElementById('sumCount').textContent=allData.length;
  document.getElementById('sumIncomeCount').textContent=allData.filter(r=>r.income>0).length+' รายการ';
  document.getElementById('sumExpenseCount').textContent=allData.filter(r=>r.expense>0).length+' รายการ';
}

function fmt(n){return n.toLocaleString('th-TH',{minimumFractionDigits:2,maximumFractionDigits:2});}

function renderTable(){
  const s=document.getElementById('searchInput').value.toLowerCase();
  const t=document.getElementById('typeFilter').value;
  const f=allData.filter(r=>(!s||r.name.toLowerCase().includes(s)||r.note.toLowerCase().includes(s))&&(!t||r.type===t));
  document.getElementById('rowCount').textContent=`แสดง ${f.length} / ${allData.length} รายการ`;
  const tbody=document.getElementById('tableBody');
  if(!f.length){tbody.innerHTML='<tr><td colspan="9"><div class="table-empty"><div class="icon">🌸</div><div>ไม่พบรายการ</div></div></td></tr>';return;}
  tbody.innerHTML=f.map((r,i)=>{
    const dc=CONFIG.DEPTS[r.deptIdx]?.color||'#f472b6';
    const dot=activeDept===0?`<span style="display:inline-block;width:7px;height:7px;border-radius:50%;background:${dc};margin-right:5px;vertical-align:middle"></span>`:'';
    return `<tr>
      <td style="color:var(--text-sub);font-size:.78rem">${i+1}</td>
      <td class="cell-date">${r.date}</td>
      <td><span class="type-badge ${r.type==='รายรับ'?'badge-in':'badge-out'}">${r.type==='รายรับ'?'💚':'❤️'} ${r.type||'-'}</span></td>
      <td class="cell-name">${dot}${r.name}</td>
      <td class="amt-in">${r.income>0?'+฿'+fmt(r.income):'<span style="color:var(--border)">—</span>'}</td>
      <td class="amt-out">${r.expense>0?'-฿'+fmt(r.expense):'<span style="color:var(--border)">—</span>'}</td>
      <td class="amt-bal">฿${fmt(r.balance)}</td>
      <td>${r.slip?`<a class="slip-btn" href="#" onclick="openSlip('${r.slip}',event)">🧾 สลิป</a>`:'<span class="no-slip">—</span>'}</td>
      <td class="cell-note">${r.note||'—'}</td>
    </tr>`;
  }).join('');
}

async function saveEntry(){
  const date=document.getElementById('eDate').value;
  const type=document.getElementById('eType').value;
  const deptI=parseInt(document.getElementById('eDept').value);
  const name=document.getElementById('eName').value.trim();
  const amount=parseFloat(document.getElementById('eAmount').value);
  const slip=document.getElementById('eSlip').value.trim();
  const note=document.getElementById('eNote').value.trim();
  if(!date||!name||isNaN(amount)||amount<=0){document.getElementById('addErr').classList.add('show');return;}
  const btn=document.getElementById('saveBtn');
  btn.textContent='⏳ กำลังบันทึก...';btn.disabled=true;
  try{
    const d=CONFIG.DEPTS[deptI];
    const income=type==='รายรับ'?amount:0,expense=type==='รายจ่าย'?amount:0;
    const params=new URLSearchParams({date,type,name,income,expense,slip,note,sheetId:d.sheetId,sheetName:d.sheetName});
    const res=await fetch(CONFIG.SCRIPT_URL+'?'+params.toString());
    const result=await res.json();
    if(result.status==='ok'){closeModal('addModal');activeDept=deptI;switchDept(deptI);}
    else throw new Error(result.message||'บันทึกไม่สำเร็จ');
  }catch(e){const el=document.getElementById('addErr');el.textContent='❌ '+(e.message||'เกิดข้อผิดพลาด');el.classList.add('show');}
  finally{btn.textContent='💾 บันทึก';btn.disabled=false;}
}
